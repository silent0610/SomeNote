给定一个字符串 `s` ，请你找出其中不含有重复字符的 **最长 子串** 的长度。

## 题解

[字符串题目](字符串题目.md)
[滑动窗口](滑动窗口.md)

```cpp
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        unordered_set<char> occ;
        int n = s.size();
        int rk = -1;
        int ans = 0;
        for(int i =0;i<n;++i)
        {
            if(i!=0)
            {
                occ.erase(s[i-1]);
                
            }
            while(rk+1<n&&!occ.count(s[rk+1]))
            {
                occ.insert(s[rk+1]);
                ++rk;
            }
            ans = max(ans,rk-i+1);
        }
        return ans;
    }
};
```
